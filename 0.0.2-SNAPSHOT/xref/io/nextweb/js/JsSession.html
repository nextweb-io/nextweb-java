199 	@Export
200 	public void getAll(final Object... params) {
201 
202 		final Object[] jsAr = params;
203 
204 		final List<BasicResult<?>> requestedEntities = new ArrayList<BasicResult<?>>(
205 				jsAr.length - 1);
206 
207 		JavaScriptObject callback_onSuccess = null;
208 		JavaScriptObject callback_onFailure = null;
209 
210 		for (int i = 0; i <= jsAr.length - 1; i++) {
211 
212 			final Object param = jsAr[i];
213 
214 			final Object gwtInstance = ExporterUtil.gwtInstance(param);
215 			// GWT.log(gwtInstance.getClass().toString());
216 			if (gwtInstance instanceof JsWrapper<?>) {
217 
218 				if (((JsWrapper) gwtInstance).getOriginal() instanceof BasicResult<?>) {
219 					// GWT.log("Adding result "
220 					// + gwtInstance.getClass().toString());
221 					requestedEntities
222 							.add((BasicResult<?>) ((JsWrapper) gwtInstance)
223 									.getOriginal());
224 				}
225 			} else {
226 				if (callback_onSuccess != null) {
227 					if (callback_onFailure != null) {
228 						throw new IllegalArgumentException(
229 								"Invalid parameters for get all. Specifiy unresolved results + one callback function.");
230 					}
231 					callback_onFailure = (JavaScriptObject) param;
232 					continue;
233 				}
234 				// GWT.log("Adding callback: " + gwtInstance.getClass());
235 				callback_onSuccess = (JavaScriptObject) param;
236 			}
237 
238 		}
239 
240 		if (callback_onSuccess == null || requestedEntities.size() == 0) {
241 			throw new IllegalArgumentException(
242 					"Invalid parameters for get all. Specifiy unresolved results + one callback function.");
243 		}
244 
245 		final JavaScriptObject callback_onSuccess_Closed = callback_onSuccess;
246 		final JavaScriptObject callback_onFailure_Closed = callback_onFailure;
247 
248 		final BasicResult[] requestedEntitiesArray = requestedEntities
249 				.toArray(new BasicResult[] {});
250 
251 		session.getAll(true, requestedEntitiesArray).get(
252 				CallbackFactory.eagerCallback(session,
253 						session.getExceptionManager(),
254 						new Closure<SuccessFail>() {
255 
256 							@Override
257 							public void apply(final SuccessFail result) {
258 								if (result.isFail()) {
259 									if (callback_onFailure_Closed == null) {
260 										session.getEngine()
261 												.getExceptionManager()
262 												.onFailure(
263 														Fn.exception(
264 																this,
265 																result.getException()));
266 										return;
267 									}
268 
269 									JH.triggerCallback(
270 											callback_onFailure_Closed,
271 											((NextwebEngineJs) session
272 													.getEngine()).jsFactory()
273 													.getWrappers(),
274 											new JavaScriptObject[] { ExporterUtil
275 													.wrap(result.getException()) });
276 									return;
277 								}
278 
279 								final List<Object> resolvedObjects = new ArrayList<Object>(
280 										requestedEntities.size());
281 								for (final BasicResult<?> requestedBasicResult : requestedEntities) {
282 									resolvedObjects.add(requestedBasicResult
283 											.get());
284 								}
285 
286 								JH.triggerCallback(callback_onSuccess_Closed,
287 										((NextwebEngineJs) session.getEngine())
288 												.jsFactory().getWrappers(),
289 										resolvedObjects.toArray());
290 							}
291 
292 						}).catchExceptions(new ExceptionListener() {
293 
294 					@Override
295 					public void onFailure(final ExceptionResult r) {
296 						if (callback_onFailure_Closed == null) {
297 							session.getEngine().getExceptionManager()
298 									.onFailure(r);
299 							return;
300 						}
301 
302 						JH.triggerCallback(callback_onFailure_Closed,
303 								((NextwebEngineJs) session.getEngine())
304 										.jsFactory().getWrappers(),
305 								new JavaScriptObject[] { ExporterUtil.wrap(r
306 										.exception()) });
307 					}
308 				}));
309 
310 	}
