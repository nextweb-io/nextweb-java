78  	public OnedbLinkList(final OnedbSession session,
79  			final ExceptionManager parentExceptionManager, final List<Link> list) {
80  		super();
81  		this.session = session;
82  		this.list = list;
83  		this.exceptionManager = session.getFactory().createExceptionManager(
84  				parentExceptionManager);
85  		this.result = session.getEngine().createResult(exceptionManager,
86  				session, new AsyncResult<NodeList>() {
87  
88  					@Override
89  					public void get(final Callback<NodeList> callback) {
90  
91  						final ListCallbackJoiner<Link, Node> joiner = new ListCallbackJoiner<Link, Node>(
92  								list, new ListCallback<Node>() {
93  
94  									@Override
95  									public void onSuccess(
96  											final List<Node> responses) {
97  										callback.onSuccess(H.factory(
98  												OnedbLinkList.this)
99  												.createNodeList(
100 														getOnedbSession(),
101 														getExceptionManager(),
102 														responses));
103 									}
104 
105 									@Override
106 									public void onFailure(final Throwable t) {
107 										callback.onFailure(Fn
108 												.exception(this, t));
109 									}
110 
111 								});
112 
113 						for (final Link link : list) {
114 
115 							final LocalCallback<Node> localCallback = joiner
116 									.createCallback(link);
117 
118 							// TODO how to deal with 'unresolvable' nodes
119 							// maybe not forward exceptions for every call?
120 							link.get(CallbackFactory.embeddedCallback(
121 									getExceptionManager(), callback,
122 									new Closure<Node>() {
123 
124 										@Override
125 										public void apply(final Node o) {
126 											localCallback.onSuccess(o);
127 										}
128 									}));
129 
130 						}
131 
132 					}
133 				});
134 	}
