321 	private String generateAddress(final Object value, final Node node,
322 			final CoreDsl dsl) {
323 		String genAddress = value.toString();
324 
325 		if (genAddress.length() > 8) {
326 			genAddress = genAddress.substring(0, 7);
327 		}
328 
329 		String cleanedString = "";
330 		for (int i = 0; i <= genAddress.length() - 1; i++) {
331 			if (!OneUtilsStrings.isSimpleCharacter(genAddress.charAt(i))) {
332 				cleanedString = cleanedString + "_";
333 			} else {
334 				cleanedString = cleanedString + genAddress.charAt(i);
335 			}
336 		}
337 
338 		cleanedString = cleanedString
339 				+ (dsl.selectFrom(dsl.reference(node.getUri()))
340 						.allChildrenFast().in(H.client(entity)).size() + 1);
341 		return cleanedString;
342 	}
