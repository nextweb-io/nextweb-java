488 				.and(new WhenCommitted() {
489

491 					public void thenDo(final WithCommittedResult r) {
492

496 					public void onFailure(final Throwable t) {
497 						session.getExceptionManager().onFailure(
498 								Fn.exception(session, t));

500

502

516 	@Override
517 	public Result<SuccessFail> getAll(final boolean asynchronous,
518 			final BasicResult<?>... results) {
519 		final Result<SuccessFail> getAllResult = session.getEngine()
520 				.createResult(session.getExceptionManager(), session,
521 						new AsyncResult<SuccessFail>() {
522 
523 							@SuppressWarnings({ "unchecked" })
524 							@Override
525 							public void get(final Callback<SuccessFail> callback) {
526 
527 								final CallbackLatch latch = new CallbackLatch(
528 										results.length) {
529 
530 									@Override
531 									public void onFailed(final Throwable arg0) {
532 										callback.onSuccess(SuccessFail
533 												.fail(arg0));
534 									}
535 
536 									@Override
537 									public void onCompleted() {
538 										callback.onSuccess(SuccessFail
539 												.success());
540 									}
541 								};
542 
543 								final BasicResult<Object>[] resultObj = (BasicResult<Object>[]) results;
544 
545 								for (final BasicResult<Object> result : resultObj) {
546 									final EagerCallback<Object> eagerCallback = CallbackFactory
547 											.eagerCallback(
548 													session,
549 													session.getExceptionManager(),
550 													new Closure<Object>() {
551 
552 														@Override
553 														public void apply(
554 																final Object o) {
555 															latch.registerSuccess();
556 														}
557 
558 													}).catchExceptions(
559 													new ExceptionListener() {
560 
561 														@Override
562 														public void onFailure(
563 																final ExceptionResult r) {
564 															latch.registerFail(r
565 																	.exception());
566 														}
567 													});
568 									result.get(eagerCallback);
569 
570 								}
571 
572 							}
573 						});
574 
575 		return getAllResult;
576 	}
